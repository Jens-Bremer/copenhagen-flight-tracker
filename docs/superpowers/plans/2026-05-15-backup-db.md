# SQLite Backup Script Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a daily SQLite backup job that writes a date-stamped copy of `data/flights.db` to `data/backups/`, retains only the 7 most recent, and sends an ntfy alert if the backup fails.

**Architecture:** A new `scripts/backup_db.py` provides `backup_database()` (side-effectful, belongs in `scripts/` per project convention). The scheduler gains a fourth job `_backup_job()` at 01:00 that wraps it in a try/except and calls `send_alert()` on failure. Two new config constants are added and validated at startup.

**Tech Stack:** Python stdlib only — `sqlite3` (live-safe `.backup()` API), `os`, `datetime`. `schedule` library for the job. `pytest` + `tmp_path` for tests.

```python
import os
import sqlite3
from datetime import date


def backup_database(db_path: str, backup_dir: str, keep_last_n: int = 7) -> str:
    """Back up db_path to backup_dir/flights_YYYY-MM-DD.db and prune old backups.

    Uses sqlite3.backup() which is safe for live databases under concurrent writes.
    Returns the path of the newly written backup file.
    """
    os.makedirs(backup_dir, exist_ok=True)
    today = date.today().isoformat()
    backup_path = os.path.join(backup_dir, f"flights_{today}.db")
    src = sqlite3.connect(db_path)
    dst = sqlite3.connect(backup_path)
    try:
        src.backup(dst)
    finally:
        dst.close()
        src.close()
    existing = sorted(
        f for f in os.listdir(backup_dir)
        if f.startswith("flights_") and f.endswith(".db")
    )
    for old in existing[:-keep_last_n]:
        os.remove(os.path.join(backup_dir, old))
    return backup_path


if __name__ == "__main__":
    import sys
    sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
    import config
    path = backup_database(config.DATABASE_PATH, config.BACKUP_DIR, config.BACKUP_KEEP_LAST_N)
    print(f"Backup written to {path}")
```

```python
def _backup_job() -> None:
    """Back up the database and alert via ntfy if the backup fails."""
    try:
        path = backup_database(
            config.DATABASE_PATH, config.BACKUP_DIR, config.BACKUP_KEEP_LAST_N
        )
        logger.info("Backup written to %s", path)
    except Exception as exc:
        logger.error("Backup failed: %s", exc)
        send_alert(
            title="Flight tracker: backup failed",
            message=str(exc),
            priority="high",
        )
```

```python
def setup_schedule() -> None:
    """Register all recurring jobs."""
    daily_time = f"{config.DAILY_WINDOW_START_HOUR:02d}:00"
    schedule.every().day.at(daily_time).do(_daily_job)
    schedule.every().day.at("01:00").do(_backup_job)
    schedule.every().day.at("23:30").do(_health_check_job)
    schedule.every().day.at("23:45").do(_csv_export_job)
    logger.info(
        "Scheduler: daily collection at %s, backup at 01:00, health check at 23:30, CSV export at 23:45",
        daily_time,
    )
```
